    # JWT的使用

    # JWT说明

    首先JWT是由header、payload和signature（签名）组成，header一般张这样

    {
  'typ': 'JWT',
  'alg': 'HS256'
}

    这部分说明了生命的类型和算法

    playload是存放有效信息的地方，一般包含三个部分

    • 标准中注册的声明
    • 公共的声明
    • 私有的声明 可以用这部分来存在用户的一些信息，例如：
    {
    UserName = "curry",
    Roles = new List<string> { "Admin", "Manager" }, 
    IsAdmin = true 
}

    jwt的第三部分是一个签证信息，这个签证信息由三部分组成：

    • header (base64后的)
    • payload (base64后的)
    • secret
    var encodedString = base64UrlEncode(header) + '.' + base64UrlEncode(payload);

var signature = HMACSHA256(encodedString, 'secret');

    # JWT使用

    安装nuget包JWT

    在Web.config文件的AppSetting节点添加

    <add key="SecureKey" value="If someone can change the world, why is that person not you?"/>

    新建几个类文件

    public class LoginRequest
{
    public string UserName { get; set; }
    public string Password { get; set; }
}

public class AuthInfo
{
    //模拟JWT的payload
    public string UserName { get; set; }
    public List<string> Roles { get; set; }
    public bool IsAdmin { get; set; }
}

public class LoginResult
{
    public bool Success { get; set; }
    public string Token { get; set; }
    public string Message { get; set; }
}

    新建控制器

    public LoginResult Post([FromBody]LoginRequest request)
{
    LoginResult rs = new LoginResult();
    //这是是获取用户名和密码的，这里只是为了模拟
    if (request.UserName == "curry" && request.Password == "123456")
    {
        AuthInfo info = new AuthInfo { UserName = "curry", Roles = new List<string> {"Admin","Manager" }, IsAdmin = true };
        try
        {
            byte[] key =Encoding.Default.GetBytes(ConfigurationManager.AppSettings["SecureKey"]);
           
            IJwtAlgorithm algorithm = new HMACSHA256Algorithm();
            IJsonSerializer serializer = new JsonNetSerializer();
            IBase64UrlEncoder urlEncoder = new JwtBase64UrlEncoder();
            IJwtEncoder encoder = new JwtEncoder(algorithm, serializer, urlEncoder);
            var token = encoder.Encode(info, key);
            
            rs.Message = "返回消息XXXX";
            rs.Token = token;
            rs.Success = true;
        }
        catch (Exception ex)
        {
            rs.Message = ex.Message;
            rs.Success = false;
        }
    }
    else
    {
        rs.Message = "fail";
        rs.Success = false;
    }
    return rs;
}

    客户端获取到token后存放在本地，在下次请求时，将token放到请求头中，这边我用postman模拟请求

    先写个特性用来统一使用

    public class ApiAuthorizeAttribute: AuthorizeAttribute
{
    /// <summary>
    /// 权限拦截器
    /// </summary>
    /// <param name="actionContext"></param>
    /// <returns></returns>
    protected override bool IsAuthorized(HttpActionContext actionContext)
    {
        var authHeader = from t in actionContext.Request.Headers where t.Key == "auth" selectt.Value.FirstOrDefault();
        if (authHeader != null)
        {
            string token = authHeader.FirstOrDefault();
            if (!string.IsNullOrEmpty(token))
            {
                try
                {
                    byte[] key =Encoding.Default.GetBytes(ConfigurationManager.AppSettings["SecureKey"]);
                    //const string secret = "If someone can change the world, why is that personnot you?";
                    //secret需要加密
                    IJsonSerializer serializer = new JsonNetSerializer();
                    IDateTimeProvider provider = new UtcDateTimeProvider();
                    IJwtValidator validator = new JwtValidator(serializer, provider);
                    IBase64UrlEncoder urlEncoder = new JwtBase64UrlEncoder();
                    IJwtDecoder decoder = new JwtDecoder(serializer, validator, urlEncoder);
                    var json = decoder.DecodeToObject<AuthInfo>(token, key, verify: true);
                    if (json != null)
                    {
                        actionContext.RequestContext.RouteData.Values.Add("auth", json);
                        return true;
                    }
                    return false;
                }
                catch (Exception)
                {
                    return false;
                }
            }
        }
        return false;
    }
}

    再创建一个控制器去测试

    public class UserController : ApiController
{
    [HttpPost]
    [Route("Get")]
    [ApiAuthorize]
    public string Get()
    {
        AuthInfo info = RequestContext.RouteData.Values["auth"] as AuthInfo;
        if (info == null)
        {
            return "获取不到，失败";
        }
        else
        {
            return $"获取到了，Auth的Name是 {info.UserName}";
        }
    }
}
