# 1、C#键值对

最经典的Dictionary<TKey, TValue>

Dictionary<int, string> dic = new Dictionary<int, string>();
            
Random rd = new Random();
for (int i = 0; i < 20; i++)
{
    int num = rd.Next(1, 20);
    if (!dic.ContainsKey(num))//键不允许重复
        dic.Add(num, "james"+num);
}
//利用KeyValuePair输出
foreach (KeyValuePair<int, string> kv in dic)
{
    Console.WriteLine(kv.Key + ":" + kv.Value);
}

Dictionary是继承了IEnumerable，Enumerable可以转成list或者数据，但同时还有两种用法：ToDictionary和ToLookup，接下来看代码

//先造一些数据
public static List<student> GetList()
{
    return new List<student>()
    {
        new student(){StuNo="0001",Grand=1,Sex="男"},
        new student(){StuNo="0002",Grand=10,Sex="男"},
        new student(){StuNo="0003",Grand=3,Sex="女"},
    };
}
public class student
{
    public string StuNo { get; set; } //学号
    public int Grand { get; set; } //年级
    public string Sex { get; set; }   //性别
}

//然后我们
List<student> stulist = GetList();
var dic1 = stulist.ToDictionary(m => m.Grand);//依据年级转成字典
foreach (KeyValuePair<int, student> item in dic1)
{
    Console.WriteLine(item.Key + ":" + item.Value.StuNo+"-"+item.Value.Sex);
}
